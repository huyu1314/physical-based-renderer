cmake_minimum_required(VERSION 3.0)

project(AwesomeRenderer)

set(CMAKE_CXX_STANDARD 14)
set(CMAKE_CXX_STANDARD_REQUIRED ON)
set(CMAKE_CXX_EXTENSIONS OFF)

set(CMAKE_MODULE_PATH ${CMAKE_MODULE_PATH} "${CMAKE_SOURCE_DIR}/cmake")


include(macros)
include(modules/sanitizers_support)
include(modules/clang_tidy_support)
include(modules/add_ide_support)

detect_platform()
print_info()
set_output_paths()

add_subdirectory_ex(3rdpart 3rd_build)
# add_subdirectory_ex(src)

SET(imgui
src/imgui/stb_textedit.h
src/imgui/stb_rect_pack.h
src/imgui/stb_truetype.h

src/imgui/imgui_internal.h
src/imgui/imconfig.h
src/imgui/imgui.cpp
src/imgui/imgui.h
src/imgui/imgui_draw.cpp
src/imgui/imgui_demo.cpp
src/imgui/imgui_user/imgui_user.cpp
src/imgui/imgui_user/imgui_user.h
src/imgui/imgui_user/imgui_tabs.cpp
src/imgui/imgui_user/imgui_tabs.h
src/imgui/imguizmo/imguizmo.cpp
src/imgui/imguizmo/imguizmo.h
)

SET(mesh
src/mesh/mesh_import.h
src/mesh/stb_image.h
)

SET(ray_tracing
src/ray_tracing/Color.h
src/ray_tracing/Math.h
src/ray_tracing/RayTracer.cpp
src/ray_tracing/RayTracer.h
src/ray_tracing/Vector.h
)

SET(GL 
src/GL/gl3w.c
src/GL/gl3w.h
src/GL/glcorearb.h
)

SET(renderer
src/renderer/the_renderer.h
src/renderer/the_renderer.cpp
)

SET(ray_tracing_in_one_weekend
src/ray_tracing_in_one_weekend
)


add_executable(AwesomeRenderer
${imgui}
${mesh}
${ray_tracing}
${renderer}
${GL}
src/imgui_impl_glfw_gl3.cpp
src/imgui_impl_glfw_gl3.h
src/main.cpp
)

target_link_libraries(AwesomeRenderer glfw)
target_link_libraries(AwesomeRenderer assimp)

link_directories(${AwesomeRenderer_SOURCE_DIR})
